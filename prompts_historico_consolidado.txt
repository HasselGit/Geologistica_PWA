class VijesService extends ChangeNotifier {
  List<Map<String, dynamic>> _viajes = [];
  bool _loading = false;
  String? _error;

  // Getters
  List<Map<String, dynamic>> get viajes => _viajes;
  bool get loading => _loading;
  String? get error => _error;

  // Constructor - puede recibir dependencies inyectadas

  late final SupabaseClient _client;

      _error = null;
      notifyListeners();

      final data = await _client
          .from('viajes')
          .select()
          .order('fecha', ascending: false);

      _loading = false;
      notifyListeners();
    }

      rethrow;
    }

import 'package:flutter/foundation.dart';
import 'package:supabase_flutter/supabase_flutter.dart';
import 'package:flutter_secure_storage/flutter_secure_storage.dart';

class AuthService extends ChangeNotifier {
  late final SupabaseClient _client;
  late final FlutterSecureStorage _storage;
  
  User? _currentUser;
  String? _error;

  // Getters
  User? get currentUser => _currentUser;
  String? get error => _error;
  bool get isLoggedIn => _currentUser != null;

  AuthService({
    SupabaseClient? client,
    FlutterSecureStorage? storage,
  }) {
    _client = client ?? Supabase.instance.client;
    _storage = storage ?? const FlutterSecureStorage();
  }

  Future<bool> login(String email, String password) async {
    try {
      _error = null;
      notifyListeners();

      final response = await _client.auth.signInWithPassword(
        email: email,
        password: password,
      );

      if (response.user != null) {
        _currentUser = response.user;
        
        // Guardar token securo
        await _storage.write(
          key: 'auth_token',
          value: response.session?.accessToken,
        );

        notifyListeners();
        return true;
      }
    } catch (e) {
      _error = 'Login failed: $e';
      notifyListeners();
      return false;
    }

    return false;
  }

  Future<void> logout() async {
    try {
      await _client.auth.signOut();
      _currentUser = null;
      await _storage.delete(key: 'auth_token');
      notifyListeners();
    } catch (e) {
      _error = 'Logout failed: $e';
      notifyListeners();
    }
  }

  Future<bool> restoreSession() async {
    try {
      final token = await _storage.read(key: 'auth_token');
      if (token != null) {
        _currentUser = _client.auth.currentUser;
        notifyListeners();
        return _currentUser != null;
      }
    } catch (e) {
      _error = 'Restore session failed: $e';
    }
    return false;
  }
}
